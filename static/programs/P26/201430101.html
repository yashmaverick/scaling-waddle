void numgen(int *a)
{

    //arrtonum(a);

    for(i=0;i<4;i++)

     for(j=0;j<=9;j++)
     {
       if(j==b[i])
         continue; 
            a[i]=j;
            if(i==0 && a[i]==0)
             continue;
            //   a[i]=(a[i]+1)%10;
            int e=a[3]+10*a[2]+100*a[1]+1000*a[0];//arrtonum(a);

         if(vis[e]!=1)

}
