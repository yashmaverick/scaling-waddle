typedef struct adj{
 long long val;
    struct adj *link;
}adj;

long long front=-1,rear=-1;
long long queue[10006];
adj *list[10006];
int sieve[10000];
void mrsieve()
{
 int i,j;
    for(i=2;i*i<=10000;i++)
 {
     if(sieve[i]==1)
     {
         for(j=2*i;j<=10000;j+=i)
         {
             sieve[j]=0;
            }
     }
 }
}
void insert(long long x)
{
 if(front==-1)
 {
     front=0;
        rear=0;
    }
 else
     rear++;
    queue[rear]=x;
}
long long delete()
{
 long long ele;
    ele=queue[front++];
    if(front>rear)
 {
     front=-1;
        rear=-1;
    }
 return ele;
}

 }

 while(testcase--)
