void push(int x)
{
 if(rear==-1)
 {
     front=0;
        rear=0;
        q[rear]=x;
    }
 else q[++rear]=x;
}
int pop()
{
 if(front==-1)
     return;
    else
 {
     return q[front++];
    }
}
