int sieve()
{
 

    int i,j;

 {
     visited[i]=0;
        prime[i]=1;
        a[i]=NULL;
    }

 for(i=2;i<=10000;i++)
     if(prime[i]==1)
         for(j=2;i*j<=10000;j++)
             prime[i*j]=0;
    j=0;

    for(i=1000;i<=10000;i++)
     if(prime[i]==1)
         nos[j++]=i;
    return j;
}
